```python
# luminoracore/tools/compiler.py
def compile_system_prompt(self, personality: Personality) -> str:
    """
    Convert personality into actual system prompt for LLMs.
    This is THE critical feature that makes personalities work.
    """
    prompt = f"You are {personality.persona.name}.\n"
    prompt += f"{personality.persona.description}\n\n"
    
    # Core traits
    prompt += "Your core personality traits:\n"
    for trait in personality.core_traits:
        prompt += f"- {trait}\n"
    
    # Linguistic profile
    prompt += f"\nYour tone is: {', '.join(personality.linguistic_profile.tone)}\n"
    prompt += f"Your formality level is: {personality.linguistic_profile.formality_level}\n"
    
    # Behavioral rules
    prompt += "\nBehavioral guidelines:\n"
    for rule in personality.behavioral_rules:
        prompt += f"- {rule}\n"
    
    # Examples
    if personality.examples:
        prompt += "\nExample interactions:\n"
        for example in personality.examples.sample_responses:
            prompt += f"User: {example.input}\nYou: {example.output}\n\n"
    
    return prompt
```
